        for(int i=0;i<MusicList.Length;i++)
        {
            yield return MusicList[i].Author;
        }
    }

    public IEnumerator<string> GetTimeEnumerator()
    {
        for(int i=0;i<MusicList.Length;i++)
        {
            yield return MusicList[i].Time;
        }
    }
}
